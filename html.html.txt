// First page submit 
nameForm.addEventListener('submit', function(event){ 
    event.preventDefault(); 
    const nameInput = document.getElementById('name').value.trim(); 
    if(nameInput !== ''){ 
        userName = nameInput; 
        nameForm.style.display = 'none'; 
        // Second page background 
        body.style.backgroundImage = "url('https://rukminim2.flixcart.com/image/480/640/xif0q/poster/x/v/9/medium-dudu-bubu-hd-poster-for-cute-baby-12x18-inch-2-posters-original-imahb2gzmfzgeguu.jpeg?q=90')";

        body.style.backgroundPosition = "center"; 
        greeting.style.display = 'block'; 
        greeting.innerHTML = `${userName}, you are here to know about us right`; 
        setTimeout(() => { 
            secondMessage.style.display = 'block'; 
            secondMessage.innerHTML = `That's great! Can you please tell the name of your boyfriend?`; 
            bfForm.style.display = 'block';

        alert('Please enter a name.'); 
    } 
});

        bfForm.style.display = 'none'; 
        secondMessage.style.display = 'none'; 
        greeting.style.display = 'none';

        body.style.background = "lightblue";
        extraPage.style.display = 'block';
        bfNameDisplay.textContent = bfName;
        
    } else { 
        alert('Please enter your boyfriend\'s name.'); 
    } 
});
